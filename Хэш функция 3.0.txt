#include <iostream>
#include <string>
#include <vector>

using namespace std;

string GetHash(string word)
{
	if (word.length() > 16 or word.length() < 8)
	{
		return "@error@";
	}
	else
	{
		string Hash;
		string Salt = "safe";
		int Size = 16;
		vector<unsigned long int> arr;
		int num[20] = { 13,23,54,756,2,34,12,87,84,456,34,23,76,89,23,54,23,54,65,32 };
		for (int i = 0; i < Size; i++)
		{
			if (i == word.length())
			{
				Size = Size - i;
				i = 0;
			}
			arr.push_back((int)word[i] * num[i] * (Size - i) * ((i - 3) *(i * i)));
		}
		for (int i = 0; i < Salt.length(); i++)
			arr.push_back((int)Salt[i] * num[i] * (Salt.length() - i) * ((i - 2) *(i * i * i)));
		for (int i = 0; i < arr.size(); i++)
		{
			if(arr[i] % 36 == 0)
				Hash.push_back((arr[i] / 18) % 10 + 48);
			else
				Hash.push_back((arr[i]) % 25 + 65);
		}
		return Hash;
	}
}
